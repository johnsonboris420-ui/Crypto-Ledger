            if fidelity < 0.95:
                self.logger.warning(
                    f"Quantum-classical discrepancy: {fidelity:.3f} fidelity",
                    extra=self.get_logger_extras(scenario.qubit_count))

                    self.anomaly_db.register_discrepancy(
                        quantum_result, 
                        classical_result,
                        scenario
                    )
            
            quantum_result.cross_validation_fidelity = fidelity
            
        except (TimeoutError, concurrent.futures.TimeoutError):
            self.logger.warning("Classical validation timed out", 
                              extra=self.get_logger_extras(scenario.qubit_count))
        except Exception as e:
            self.logger.error(f"Classical validation failed: {str(e)}", 
                             extra=self.get_logger_extras(scenario.qubit_count))

                self.trigger_graceful_shutdown(
                    reason=f"Persistent quantum instability (ECC L{self.error_correction_level}): {result.integrity_score:.2f}"
                )
                return True
        else:
            self.consecutive_failures = 0

        if result.state_fidelity < collapse_threshold:
            self.logger.critical("Quantum state collapse detected!", 
                               extra=self.get_logger_extras(result.qubit_count))
            self.trigger_graceful_shutdown(reason="Quantum state collapse")
            return True
            
        return False

    def calculate_composite_stability(self, metrics):
        weights = {
            'integrity': 0.4,
            'fidelity': 0.3,
            'decoherence': 0.2,
            'gate_error': 0.1
        }
        
        stability = 0
        for metric, values in metrics.items():
            tau, _ = kendalltau(range(len(values)), values)
            trend_stability = (1 - abs(tau)) / 2
            volatility = np.std(values) / np.mean(values) if np.mean(values) > 0 else 1
            metric_stability = 0.7 * trend_stability + 0.3 * (1 - min(1, volatility))
            stability += weights[metric] * (1 - metric_stability)
            
        return min(1.0, stability)

    def adjust_safety_threshold(self):
        if len(self.test_history) < self.STABILITY_WINDOW:
            return
            
        recent_scores = [r.integrity_score for r in 
                         list(self.test_history)[-self.STABILITY_WINDOW:]]
        avg_score = np.mean(recent_scores)
        decoherence_rates = [r.decoherence_rate for r in 
                            list(self.test_history)[-self.STABILITY_WINDOW:]]
        stability = np.mean(decoherence_rates) * 2 + np.std(recent_scores) / avg_score
        
        if stability > 0.25:
            new_threshold = min(0.55, 
                self.safety_threshold + stability * self.THRESHOLD_HYSTERESIS
            )
            self.logger.info(f"Adjusting safety threshold ↑ {self.safety_threshold:.3f}→{new_threshold:.3f}", 
                           extra=self.get_logger_extras())
            self.safety_threshold = new_threshold
        elif stability < 0.1 and self.safety_threshold > 0.4:
            new_threshold = max(0.35, 
                self.safety_threshold - 0.5 * self.THRESHOLD_HYSTERESIS
            )
            self.logger.info(f"Adjusting safety threshold ↓ {self.safety_threshold:.3f}→{new_threshold:.3f}", 
                           extra=self.get_logger_extras())
            self.safety_threshold = new_threshold

            if len(values) > 1:
                x = np.arange(len(values))
                slope, _, r_value, _, _ = linregress(x, values)
                metrics[metric] = {
                    'slope': slope,
                    'r_squared': r_value**2,
                    'last_5_avg': np.mean(values[-5:])
                }
        
        ent_values = [r.entanglement_entropy for r in recent_results if hasattr(r, 'entanglement_entropy')]
        if len(ent_values) > 5:
            x = np.arange(len(ent_values))
            slope, _, r_value, _, _ = linregress(x, ent_values)
            metrics['entanglement_entropy'] = {
                'slope': slope,
                'r_squared': r_value**2,
                'last_5_avg': np.mean(ent_values[-5:])
            }
        
        return metrics

    def calculate_resource_utilization(self):
        if not self.test_history:
            return {}
        
        qubit_counts = [r.qubit_count for r in self.test_history]
        durations = [r.execution_time for r in self.test_history]
        ecc_levels = [r.ecc_level if hasattr(r, 'ecc_level') else 0 for r in self.test_history]
        resource_consumption = [q * t for q, t in zip(qubit_counts, durations)]
        
        return {
            'total_qubit_seconds': sum(resource_consumption),
            'avg_qubits': np.mean(qubit_counts),
            'max_qubits': max(qubit_counts),
            'avg_duration': np.mean(durations),
            'avg_ecc_level': np.mean(ecc_levels),
            'ecc_overhead': self.calculate_ecc_overhead()
        }

    def calculate_ecc_overhead(self):
        if not self.quantum_error_correction or not self.test_history:
            return 0
        
        ecc_results = [r for r in self.test_history if r.ecc_level > 0]
        no_ecc_results = [r for r in self.test_history if r.ecc_level == 0]
        
        if not ecc_results or not no_ecc_results:
            return 0
        
        ecc_qubits = np.mean([r.qubit_count for r in ecc_results])
        no_ecc_qubits = np.mean([r.qubit_count for r in no_ecc_results])
        ecc_time = np.mean([r.execution_time for r in ecc_results])
        no_ecc_time = np.mean([r.execution_time for r in no_ecc_results])
        
        qubit_overhead = (ecc_qubits - no_ecc_qubits) / no_ecc_qubits
        time_overhead = (ecc_time - no_ecc_time) / no_ecc_time
        
        return {
            'qubit_overhead': qubit_overhead,
            'time_overhead': time_overhead,
            'composite_overhead': 0.7 * qubit_overhead + 0.3 * time_overhead
        }

    def calculate_ecc_effectiveness(self):
        if not self.quantum_error_correction or len(self.test_history) < 20:
            return {}
        
        ecc_levels = {}
        for result in self.test_history:
            if hasattr(result, 'ecc_level'):
                level = result.ecc_level
                ecc_levels.setdefault(level, []).append(result)
        
        effectiveness = {}
        for level, results in ecc_levels.items():
            if level == 0 or len(results) < 5:
                continue
                
            logical_errors = np.array([r.logical_error_rate for r in results])
            physical_errors = np.array([r.physical_error_rate for r in results])
            
            suppression_factor = 0
            if np.mean(physical_errors) > 0:
                suppression_factor = np.mean(physical_errors) / np.mean(logical_errors)
            
            fidelity_improvement = 0
            if 0 in ecc_levels and len(ecc_levels[0]) >= 5:
                no_ecc_fidelity = np.mean([r.state_fidelity for r in ecc_levels[0]])
                ecc_fidelity = np.mean([r.state_fidelity for r in results])
                fidelity_improvement = ecc_fidelity - no_ecc_fidelity
            
            conf_interval = 1.96 * np.std(logical_errors) / np.sqrt(len(results))
            
            effectiveness[level] = {
                'suppression_factor': suppression_factor,
                'fidelity_improvement': fidelity_improvement,
                'avg_logical_error': np.mean(logical_errors),
                'error_confidence': conf_interval,
                'sample_size': len(results)
            }
        
        return effectiveness

    def calculate_stability_score(self, report):
        if not report or 'trend_analysis' not in report or not report['trend_analysis']:
            return 0.0
        
        weights = {
            'integrity_score': 0.35,
            'state_fidelity': 0.30,
            'decoherence_rate': 0.20,
            'entanglement_entropy': 0.15
        }
        
        score = 0.0
        total_weight = 0.0
        
        for metric, weight in weights.items():
            if metric in report['trend_analysis']:
                data = report['trend_analysis'][metric]
                trend_factor = max(0, 1 - np.tanh(-data['slope'] * 100))
                recent_factor = min(1.0, data['last_5_avg'] / 0.8) if metric != 'decoherence_rate' else min(1.0, 0.2 / data['last_5_avg'])
                metric_score = 0.6 * trend_factor + 0.4 * recent_factor
                score += weight * metric_score
                total_weight += weight
        
        if 'environment_stability' in report:
            env_score = min(1.0, report['environment_stability'] / 0.95)
            score = 0.8 * score + 0.2 * env_score
        
        if total_weight > 0:
            score /= total_weight
        
        return max(0.0, min(1.0, score))

    def track_entanglement_evolution(self):
        if len(self.test_history) < 4:
            return
        
        entropies = [r.entanglement_entropy for r in self.test_history 
                    if hasattr(r, 'entanglement_entropy')]
        
        if len(entropies) < 4:
            return
        
        propagation_factors = [entropies[i]/entropies[i-1] 
                              for i in range(1, len(entropies)) 
                              if entropies[i-1] > 0]
        
        if not propagation_factors:
            return
            
        avg_propagation = np.mean(propagation_factors)
        propagation_std = np.std(propagation_factors)
        recent_entropies = entropies[-4:]
        dist_variance = np.var(recent_entropies)
        
        darwinism_factor = 0.0
        if len(entropies) > 10:
            darwinism_factor = kendalltau(entropies, range(len(entropies)))[0]
        
        if dist_variance > 0.5:
            self.logger.warning(f"High entanglement variance: {dist_variance:.3f}", 
                              extra=self.get_logger_extras())
        
        if propagation_std > 0.3:
            self.logger.warning(f"Unstable entanglement propagation: σ={propagation_std:.3f}", 
                              extra=self.get_logger_extras())
        
        if darwinism_factor < -0.4:
            self.logger.info("Quantum Darwinism pattern detected", 
                           extra=self.get_logger_extras())
        
        update_entanglement_models({
            'avg_propagation': avg_propagation,
            'propagation_std': propagation_std,
            'distribution_variance': dist_variance,
            'darwinism_factor': darwinism_factor
        })

    def detect_quantum_anomaly_correlation(self):
        if len(self.test_history) < 20:
            return
        
        anomalies = []
        temperatures = []
        decoherences = []
        fidelities = []
        entropies = []
        
        for result in self.test_history:
            anomalies.append(1 if result.anomaly_signature else 0)
            temperatures.append(result.temperature)
            decoherences.append(result.decoherence_rate)
            fidelities.append(result.state_fidelity)
            if hasattr(result, 'entanglement_entropy'):
                entropies.append(result.entanglement_entropy)
        
        if len(entropies) < len(anomalies):
            entropies = entropies + [np.mean(entropies)] * (len(anomalies) - len(entropies))
        
        data = np.array([anomalies, temperatures, decoherences, fidelities, entropies])
        corr_matrix = np.corrcoef(data)
        
        significant_correlations = {}
        for i, param in enumerate(['anomalies', 'temperature', 'decoherence', 'fidelity', 'entropy']):
            for j, param2 in enumerate(['anomalies', 'temperature', 'decoherence', 'fidelity', 'entropy']):
                if i < j and abs(corr_matrix[i, j]) > 0.5:
                    significant_correlations[f"{param}-{param2}"] = corr_matrix[i, j]
        
        if significant_correlations:
            self.anomaly_db.update_correlation_map(significant_correlations)
            self.logger.info(f"Detected correlations: {significant_correlations}", 
                           extra=self.get_logger_extras())

    def apply_quantum_error_mitigation(self, scenario):
        noise_profile = self.quantum_computer.get_current_noise_profile()
        strategy = self.select_mitigation_strategy(noise_profile)
        return apply_quantum_error_mitigation(scenario, strategy)
    
    def select_mitigation_strategy(self, noise_profile):
        if noise_profile.t1_dominant:
            return "T1_FOCUSED_MITIGATION"
        elif noise_profile.high_crosstalk:
            return "CROSSTALK_SUPPRESSION"
        elif noise_profile.gate_error_dominant:
            return "GATE_ERROR_CORRECTION"
        else:
            return "STANDARD_MITIGATION"
